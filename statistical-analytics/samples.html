R

Data Profile

Use before modeling to confirm the row count, target availability, missingness, and duplicate rows.

DATA <- read.csv("your_data.csv")
TARGET <- "target"

data.frame(
  rows = nrow(DATA),
  columns = ncol(DATA),
  missing_target = sum(is.na(DATA[[TARGET]])),
  duplicate_rows = sum(duplicated(DATA))
)
sort(colMeans(is.na(DATA)), decreasing = TRUE)

Train/Test Baseline

Use when prediction quality matters and you need a first holdout score.

from sklearn.model_selection import train_test_split
from sklearn.pipeline import make_pipeline
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LinearRegression

X = DATA[["x1", "x2", "x3"]]
y = DATA["TARGET"]
X_train, X_test, y_train, y_test = train_test_split(X, y, random_state=123)
model = make_pipeline(StandardScaler(), LinearRegression()).fit(X_train, y_train)
print(model.score(X_test, y_test))

Model-Ready Feature View

Use SQL to lock the row grain and produce a stable modeling table.

CREATE OR REPLACE VIEW model_features AS
SELECT
  id_col,
  target,
  x1,
  x2,
  x1 * x2 AS x1_x2_interaction,
  CASE WHEN row_number() OVER (ORDER BY id_col) % 5 = 0
       THEN 'test' ELSE 'train' END AS split
FROM data_table
WHERE target IS NOT NULL;

Prompt Evaluation Summary

Use for prompt, model, cost, factuality, and latency comparisons.

SELECT
  prompt_id,
  model_id,
  COUNT(*) AS outputs_evaluated,
  AVG(score) AS mean_score,
  AVG(factual) AS factuality_rate,
  AVG(confidence) AS mean_confidence,
  AVG(cost_usd) AS mean_cost_usd
FROM llm_evaluations
GROUP BY prompt_id, model_id;
