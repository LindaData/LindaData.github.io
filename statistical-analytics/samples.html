Python Samples

Data profile 
import pandas as pd

DATA = pd.read_csv("your_data.csv")
print(DATA.info())
print(DATA.describe(include="all"))
print(DATA.isna().mean())
OLS 
import statsmodels.formula.api as smf

fit = smf.ols("TARGET ~ x1 + x2 + x3", data=DATA).fit()
print(fit.summary())
Logistic regression 
fit = smf.logit("target_binary ~ x1 + x2 + x3", data=DATA).fit()
print(fit.summary())
DATA["predicted_probability"] = fit.predict(DATA)
Train/test validation 
from sklearn.model_selection import train_test_split
from sklearn.pipeline import make_pipeline
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LinearRegression

X = DATA[["x1", "x2", "x3"]]
y = DATA["TARGET"]
X_train, X_test, y_train, y_test = train_test_split(X, y, random_state=123)
model = make_pipeline(StandardScaler(), LinearRegression()).fit(X_train, y_train)
print(model.score(X_test, y_test))
PCA and clustering 
from sklearn.decomposition import PCA
from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler

X = StandardScaler().fit_transform(DATA[["x1", "x2", "x3"]])
components = PCA(n_components=2).fit_transform(X)
clusters = KMeans(n_clusters=3, n_init="auto", random_state=123).fit_predict(X)
LLM calibration 
LLM_DATA["confidence_bin"] = pd.qcut(LLM_DATA["confidence"], q=5)
calibration = LLM_DATA.groupby("confidence_bin", observed=True).agg(
    mean_confidence=("confidence", "mean"),
    observed_accuracy=("factual", "mean"),
    n=("factual", "size"),
)
brier = ((LLM_DATA["factual"] - LLM_DATA["confidence"]) ** 2).mean()

SQL Samples

Profile table 
SELECT
  COUNT(*) AS rows,
  COUNT(*) - COUNT(target) AS missing_target,
  AVG(target) AS mean_target,
  STDDEV_SAMP(target) AS sd_target
FROM data_table;
Regression-ready features 
CREATE OR REPLACE VIEW model_features AS
SELECT
  target,
  x1,
  x2,
  x1 * x2 AS x1_x2_interaction,
  CASE WHEN row_number() OVER (ORDER BY id_col) % 5 = 0
       THEN 'test' ELSE 'train' END AS split
FROM data_table
WHERE target IS NOT NULL;
A/B summary 
SELECT
  variant,
  COUNT(*) AS n,
  SUM(converted) AS conversions,
  AVG(converted) AS conversion_rate
FROM experiment_table
GROUP BY variant;
LLM prompt summary 
SELECT
  prompt_id,
  model_id,
  COUNT(*) AS outputs_evaluated,
  AVG(score) AS mean_score,
  AVG(factual) AS factuality_rate,
  AVG(confidence) AS mean_confidence,
  AVG(cost_usd) AS mean_cost_usd
FROM llm_evaluations
GROUP BY prompt_id, model_id;
